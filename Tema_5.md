```python
import math

one = [12, 25, 3, 48, 71]
two = [5, 18, 40, 62, 98]
three = [4, 21, 37, 56, 84]

max_sides = [max(one), max(two), max(three)]
min_sides = [min(one), min(two), min(three)]

def triangle_area(a, b, c):
    s = (a + b + c) / 2
    return math.sqrt(s * (s - a) * (s - b) * (s - c))

area_max = triangle_area(*max_sides)
area_min = triangle_area(*min_sides)

print(f"Площадь треугольника из максимальных сторон: {area_max}")
print(f"Площадь треугольника из минимальных сторон: {area_min}")
```

• max(list): Возвращает максимальный элемент из списка.
• min(list): Возвращает минимальный элемент из списка.
• triangle_area(a, b, c): Вычисляет площадь треугольника по трем сторонам (a, b, c) с использованием формулы Герона. 
• math.sqrt(x): Вычисляет квадратный корень из числа x.

def create_set(lst):
    result_set = set()
    counts = {}
    for num in lst:
        if num in counts:
            counts[num] += 1
        else:
            counts[num] = 1

    for num, count in counts.items():
        result_set.add(num)
        if count > 1:
            for i in range(2, count + 1):
                result_set.add(str(num) * i) 

    return result_set

list_1 = [1, 1, 3, 3, 1]
list_2 = [5, 5, 5, 5, 5, 5, 5]
list_3 = [2, 2, 1, 2, 2, 5, 6, 7, 1, 3, 2, 2]

print(create_set(list_1))
